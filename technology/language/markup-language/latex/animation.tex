\documentclass{ctexart}

\usepackage{tikz}
% 动画包
% autoplay：页面打开后开始动画。 还恢复播放先前暂停的动画。
% autopause：页面关闭时暂停动画，而不是停止并将其倒回到默认帧。
% nomouse：动画中禁用鼠标控制，默认是开启鼠标控制，鼠标移入动画区域，会有一层框架，双击会暂停动画
% loop：动画到达终点后立即重新开始。
\usepackage[
	autoplay,
	autopause,
	nomouse,
	loop
]{animate}
