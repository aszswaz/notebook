\documentclass{ctexart}

\usepackage{tikz}
% 动画包
% autoplay：页面打开后开始动画。 还恢复播放先前暂停的动画。
% autopause：页面关闭时暂停动画，而不是停止并将其倒回到默认帧。
% nomouse：动画中禁用鼠标控制，默认是开启鼠标控制，鼠标移入动画区域，会有一层框架，双击会暂停动画
% loop：动画到达终点后立即重新开始。
\usepackage[
	autoplay,
	autopause,
	nomouse,
	loop
]{animate}

\begin{document}

The first animation.

% \begin{animateinline}[begin={...}, end={\end{...}}]{frame rate}
% begin 和 end 是设置要使用的环境，frame rate 是每秒播放的帧
% \useasboundingbox (0, 0) rectangle(4, 4); 的作用类似画布，指定动画的范围
\begin{animateinline}[
		begin={
				\begin{tikzpicture}[black, scale = 1,line width = 1pt]
					\useasboundingbox (0, 0) rectangle(4, 4);
					},
					end={\end{tikzpicture}}
	]{2}
	% \newframe 表示创建新的帧，到再次遇到这个指令之前，所有绘制的推向都是这一帧的内容
	\newframe;
	\draw[->] (0, 0) -- (1, 1);
	\draw[->] (0, 0) -- (1, 0);
	\newframe \draw[->](0, 0) -- (2, 2);
	\newframe \draw[->](0, 0) -- (3, 3);
	\newframe \draw[->](0, 0) -- (4, 4);
\end{animateinline}

\end{document}
